## 5. wp 로그인을 firebase Auth 적용하기. 
function.php를 신설하기.
/**
 * 워드프레스 로그인 시 Firebase Auth와 대조하는 로직
 */
add_filter('authenticate', function($user, $username, $password) {
    // 1. 아이디나 비밀번호가 비어있으면 기본 로직으로 패스
    if (empty($username) || empty($password)) return $user;

    // 2. Firebase Admin SDK 또는 REST API를 호출하여 비밀번호 검증
    // (여기서는 사용자가 입력한 비밀번호가 GW의 규칙과 맞는지 체크)
    $is_valid_in_gw = check_firebase_auth($username, $password); 

    if ($is_valid_in_gw) {
        $user = get_user_by('login', $username);
        
        // 3. 만약 WP에 유저가 없다면 즉석에서 생성 (비번 없이)
        if (!$user) {
            $user_id = wp_create_user($username, wp_generate_password(), $username . '@example.com');
            $user = get_user_by('id', $user_id);
        }
        return $user; // 로그인 승인
    }

    return $user; // 실패 시 기본 워드프레스 인증으로 넘어감
}, 20, 3);
